\begin{enumerate}[(a)]
\item For a non-sufferer, $X\sim N(7,9)$,

Compute $P(X \geq 10)$

%----------------%
\begin{framed}

\noindent \textbf{Z Score for $X = 10$}

\[z_{10}  = \frac{10-7}{ 3}  = \frac{3}{3} = 1\]
\end{framed}
%------------------%
\begin{eqnarray*}
P(x \geq 10 ) &=&  1- P(X \leq 10)
\\ & &
\\ &=&  1 - P(Z \leq 1)
\\ & & 
\\ &=& 1 - 0.8413
\\ & &
\\ &=& 0.1587 \\
\end{eqnarray*}
where $Z\sim N(0,1)$
